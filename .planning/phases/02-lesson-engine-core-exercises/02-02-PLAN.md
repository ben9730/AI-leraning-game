```typescript
export type ExerciseType = 'mcq' | 'free-text' | 'pick-better' | 'fill-blank' | 'spot-problem' | 'simulated-chat'

export interface MCQExercise extends ExerciseBase {
  type: 'mcq'
  options: LocalizedString[]
  correctIndex: number
  explanation: LocalizedString
}

export interface FreeTextExercise extends ExerciseBase {
  type: 'free-text'
  starterPrompt?: LocalizedString
  simulatedResponse?: LocalizedString
  rubric: PromptRubric
  modelAnswer: LocalizedString
  positiveFeedback: LocalizedString
  improvementFeedback: LocalizedString
}

export interface PickBetterExercise extends ExerciseBase {
  type: 'pick-better'
  optionA: LocalizedString
  optionB: LocalizedString
  betterOption: 'A' | 'B'
  explanation: LocalizedString
}

export interface FillBlankExercise extends ExerciseBase {
  type: 'fill-blank'
  template: LocalizedString          // Contains "___" as blank marker
  acceptableAnswers: LocalizedString[]
  explanation: LocalizedString
}

export interface SpotProblemExercise extends ExerciseBase {
  type: 'spot-problem'
  problematicPrompt: LocalizedString
  issues: LocalizedString[]           // Correct answers
  distractors: LocalizedString[]      // Wrong answers
  hints: LocalizedString[]
  explanation: LocalizedString
}

export interface PromptRubric {
  criteria: RubricCriterion[]
  passingScore: number                          // 0-100
  scoringMethod: 'checklist' | 'weighted-keyword'
}

export interface RubricCriterion {
  key: 'clarity' | 'specificity' | 'context' | 'intent'
  label: LocalizedString
  weight: number
  keywords: string[]
  required: boolean
}
```

  <action>
    Write all test files FIRST (RED phase), then implement evaluators (GREEN phase).

    All evaluators are **pure functions** — no React, no hooks, no side effects. Each takes the typed exercise + user answer and returns `EvaluationResult`.

    **evaluateMCQ(exercise: MCQExercise, selectedIndex: number): EvaluationResult**
    - Binary: correct index = 100, wrong = 0
    - feedback is always exercise.explanation (shown regardless of correctness)

    **evaluatePickBetter(exercise: PickBetterExercise, selectedOption: 'A' | 'B'): EvaluationResult**
    - Binary: matches betterOption = 100, wrong = 0
    - feedback is exercise.explanation

    **evaluateFreeText(exercise: FreeTextExercise, answer: string, lang: 'en' | 'he'): EvaluationResult**
    - For English: normalize with `answer.trim().toLowerCase()`
    - For Hebrew: normalize with `answer.trim()` only (no toLowerCase — Hebrew has no case)
    - If `scoringMethod === 'checklist'`: equal weight per criterion (100 / criteria.length each)
    - If `scoringMethod === 'weighted-keyword'`: use `criterion.weight * 100` for each
    - For each criterion: check if normalized answer contains any keyword (keywords also normalized same way)
    - If any criterion has `required: true` and is not met, set `passed = false` regardless of total score
    - Otherwise: `passed = score >= exercise.rubric.passingScore`
    - feedback: positiveFeedback if passed, improvementFeedback if not
    - breakdown: `Record<string, number>` mapping criterion.key to its score contribution

    **evaluateFillBlank(exercise: FillBlankExercise, answer: string, lang: 'en' | 'he'): EvaluationResult**
    - Normalize answer: English = `trim().toLowerCase()`, Hebrew = `trim()`
    - Normalize each acceptableAnswer the same way using `answer[lang]`
    - Match if normalized answer includes any acceptable answer OR any acceptable answer includes normalized answer
    - Binary: 100 if match, 0 if not
    - feedback: exercise.explanation

    **evaluateSpotProblem(exercise: SpotProblemExercise, selectedIndices: number[], lang: 'en' | 'he'): EvaluationResult**
    - Build combined list: issues (indices 0..issues.length-1) then distractors (indices issues.length..end)
    - Count correct selections: selections that are in the issues range
    - Count distractor selections: selections that are in the distractors range
    - Score = (correctCount / issues.length) * 100 - (distractorCount * 25), floored at 0, rounded
    - passed = score >= 50
    - feedback: exercise.explanation

    - On "Continue": call `onComplete(result)`
    - CRITICAL: Do NOT call `onComplete` immediately on submit — user must see feedback first
    - All layout: `paddingStart/End`, `marginStart/End`, NO `left`/`right`
    - All text content: read from `exercise.field[lang]` where `lang = i18n.language as 'en' | 'he'`
